@login_required
def dashboard_view(request):
    """
    Affiche la liste de tous les produits avec leur stock actuel
    et les produits en alerte.
    """

    # Optimisation: Calcule le stock de TOUS les produit en 1 seule requete
    # 'stock_total' = la saomme de 'mouvement_quantite'.
    # Coalesce (..., 0) remplace None par 0 si un produit n'a aucun mouvement
    produits_avec_stock = Produit.objects.annotate(
        stock_total=Coalesce(Sum('mouvements__quantite'), 0)
    ).order_by('designation')

    # Filtre les produit en alerte en
    produits_en_alerte = [
        p for p in produits_avec_stock if p.stock_total <= p.seuil_alerte
    ]

    context = {
        'produits': produits_avec_stock,
        'produits_en_alerte': produits_en_alerte,
    }

    return render(request, 'stock/dashboard.html', context)

@login_required
def produit_detail_view(request, pk):
    """
    Affiche l'information sur un produit spécifique (grâce à sa 'pk' ou ID)
    et le formulaire pour faire un mouvement.
    """
    # Récupère le produit ou renvoie une erreur 404 s'il n'existe pas
    produit = get_object_or_404(
        Produit.objects.annotate(
            stock_total=Coalesce(Sum('mouvements__quantite'), 0)
        ), 
        pk=pk
    )
    
    # Récupère la liste des équipes pour le formulaire de "Livraison"
    equipes = Equipe.objects.all()
    
    # Récupère les 5 derniers mouvements pour ce produit
    derniers_mouvements = MouvementStock.objects.filter(produit=produit).order_by('-date_mouvement')[:5]

    context = {
        'produit': produit,
        'equipes': equipes,
        'derniers_mouvements': derniers_mouvements,
    }
    # On enverra ces données au template 'stock/produit_detail.html'
    return render(request, 'stock/produit_detail.html', context)

@login_required
def creer_mouvement_view(request):
    """
    Cette vue ne gère que les requêtes POST du formulaire.
    Elle n'a pas de template HTML, elle traite les données et redirige.
    """
    if request.method == 'POST':
        # Récupération des données du formulaire
        produit_id = request.POST.get('produit_id')
        type_mouvement = request.POST.get('type_mouvement')
        quantite_str = request.POST.get('quantite')
        equipe_id = request.POST.get('equipe')
        document_ref = request.POST.get('document_ref')

        # --- Validation et logique métier ---
        produit = get_object_or_404(Produit, pk=produit_id)
        
        try:
            quantite = int(quantite_str)
            if quantite <= 0:
                raise ValueError("La quantité doit être positive")
        except (ValueError, TypeError):
            messages.error(request, "Quantité invalide.")
            return redirect('produit-detail', pk=produit_id)

        # Règle de gestion : les sorties doivent être négatives dans la BDD
        quantite_a_enregistrer = quantite
        if type_mouvement in ['LIVRAISON', 'CASSE', 'RETOUR_FOURNISSEUR']:
            quantite_a_enregistrer = -abs(quantite)
            
            # Scénario 2 (Cas d'erreur) : Vérification du stock suffisant
            stock_actuel = produit.stock_actuel # Utilise la @property du modèle
            if stock_actuel + quantite_a_enregistrer < 0:
                messages.error(request, f"Stock insuffisant ! Stock actuel : {stock_actuel}.")
                return redirect('produit-detail', pk=produit_id)
        
        # Gestion de l'équipe (si c'est une livraison)
        equipe = None
        if type_mouvement == 'LIVRAISON' and equipe_id:
            equipe = get_object_or_404(Equipe, pk=equipe_id)
        
        # Création du mouvement
        MouvementStock.objects.create(
            produit=produit,
            type_mouvement=type_mouvement,
            quantite=quantite_a_enregistrer,
            employe=request.user, # Règle : on lie le mouvement à l'utilisateur connecté
            equipe=equipe,
            document_ref=document_ref
        )

        messages.success(request, "Mouvement enregistré avec succès.")
        # Redirige vers la page du produit qui vient d'être modifié
        return redirect('produit-detail', pk=produit_id)
    
    # Si la méthode n'est pas POST, rediriger vers le dashboard
    return redirect('dashboard')


# --- Vue 4 : Historique des Mouvements ---
@login_required
def historique_mouvements_view(request):
    """
    Affiche l'historique des mouvements.
    Implémente la règle de permission :
    - L'admin voit TOUT.
    - L'utilisateur standard ne voit que SES mouvements.
    """
    # Optimisation : 'select_related' charge les infos du produit, 
    # de l'employé et de l'équipe en une seule requête BDD.
    base_query = MouvementStock.objects.select_related(
        'produit', 'employe', 'equipe'
    ).order_by('-date_mouvement') # Du plus récent au plus ancien

    # Règle de permission :
    if request.user.is_superuser:
        # Un admin voit tous les mouvements
        mouvements = base_query.all()
    else:
        # Un utilisateur standard ne voit que ses propres mouvements
        mouvements = base_query.filter(employe=request.user)

    context = {
        'mouvements': mouvements
    }
    return render(request, 'stock/historique_mouvements.html', context)

@login_required
def produit_list_view(request):
    """
    Affiche la liste de tous les produits, avec recherche
    """
    # Récupère le terme de recherche depuis l'URL (ex: ?q=vanne)
    query = request.GET.get('q', '')

    produits = Produit.objects.all().order_by('designation')

    if query:
        # Filtre si un terme de recherche est fourni
        produits = produits.filter(
            Q(reference__incontains=query) | 
            Q(designation__icontains=query)
        )
    context = {
        'produits': produits,
        'query': query,
    }
    return render(request, 'stock/produit_list.html', context)

@login_required
def produit_form_view(request, pk=None):
    """
    Gère la création (si pk=None) et la modification (si pk est fourni)
    d'un produit
    """
    instance = None
    if pk:
        # C'est une modification, on charge le produit existant
        instance = get_object_or_404(Produit, pk=pk)
    
    # On passe l'instance au formulaire. Si instance=None, le formulare est vide.
    form = ProduitForm(request.POST or None, instance=instance)

    if request.method == 'POST' and form.is_valid():
        form.save()
        messages.success(request, f"Produit '{instance.designation}' enregisté.")
        return redirect('produit-list') # Redirige vers la liste
    
    context = {
        'form': form,
        'instance': instance, # Pour savoir si on est en "Création" ou "Modificaiton"
    }
    return render(request, 'stock/produit_form.html', context)
